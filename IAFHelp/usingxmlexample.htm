Using XML - example
   

Read a database table and store it into the XML file.



$xmlstat = xml.Open("xmldir:outfile.xml")
if $xmlstat = 0
      $xmlstat = xml.Writetext('<?xml version="1.0" encoding="utf-8" ?>') !! XML declaration

      $xmlstat = xml.Setxmldecpoint('.')

      $xmlstat = xml.Writenode('records')

      select_all * from tbl1 where [condition]
            $xmlstat = xml.Writenode('record')
            
            $xmlstat = xml.Writeelement('alpha_1',alpha_field)
            $xmlstat = xml.Writeelement('num_1',num1_field,"ZZZ9")
            $xmlstat = xml.Writeelement('num_2',num2_field,"ZZZ9,99")
            $xmlstat = xml.Writeelement('date_1',date_field,'YYYY-MM-DD')
            $utf8 = convert(alpha_field_1,1250,65001)
            $xmlstat = xml.Writeelement('alpha_2',$utf8)

            $xmlstat = xml.Writeendnode('record')
      end_select

      $xmlstat = xml.Writeendnode('records')

      $xmlstat = xml.Close()

else
      Pause "Open file error" | $xmlstat
end_if

This piece of code downloads a file from FTP server.

********************************************************
$ftpmsg is an alpha variable, length = 255 bytes.
********************************************************

$ftpmsg = Ftp.Open('myserver','user1','pwd1')   !! mode=active
if $ftpmsg <> SPACE
      Pause "Ftp error: " | $ftpmsg
else
      !! Root directory on FTP server is '/'
      if ftp.Getcurrentdirectory() <> '/'
            $ftpmsg = ftp.Setcurrentdirectory('/')    !! set current dir = ROOT
      end_if
      !! download file from root directory
      $ftpmsg = Ftp.Downloadfile('remotefile.jpg','c:\temp\image1.jpg','O','progress_1')
      Ftp.Close()
end_if

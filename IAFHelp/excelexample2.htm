Using Excel - Example 2
   

This piece of code transfers data from database to the file "c:\dat\Test.xls".

******************************
$estat is numeric variable.
******************************

$estat = excel.Open()    !!create new, empty document
if $estat <> 0
            Pause "Excel error: " | $estat
else
            init counter_1
            read_all file1
                        add 1 to counter_1
                        $estat = excel.Setcell(counter_1,1,fld1,1250) !!alpha field, codepage=1250
                        $estat = excel.Setcell(counter_1,2,fld2)
                        $estat = excel.Setcell(counter_1,3,fld3)
                        $estat = excel.Setcell(counter_1,4,add_days(fld4,5))
            end_read
            $estat = excel.Run("save_exit", "c:\dat\Test.xls")
end_if
